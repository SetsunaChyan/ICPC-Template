	dp[i][0]=a[i];
for(int j=1;j<=30;++j){
	for(int i=1;i+(1LL<<(j-1))<=n;++i){
		dp[i][j]=max(dp[i][j-1],dp[i+(1LL<<(j-1))][j-1]);//min
	}
}

int ask(int l,int r){
	int k=(int)log2(r-l+1);
	return max(dp[l][k],dp[r-(1LL<<k)+1][k]);//min
}
