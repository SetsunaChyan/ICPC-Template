{
    double lm=(l+r)/2,rm=(lm+r)/2;
    if(judge(lm)>judge(rm)) l=lm;
    else r=rm;
}

while(l+1<r)
{
    int lm=(l+r)>>1,rm=(lm+r)>>1;
    if(judge(lm)>judge(rm)) l=lm;
    else r=rm;
}
